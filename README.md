# conda 가상환경 만들기
```
# 새로운 가상환경 만들기
conda create -n llm_env python=3.10

 # 가상환경 목록 확인
conda env list

# eda_env 가상환경 활성화
conda activate eda_env

# jupyter lab 실행
jupyter lab


# 현재 디렉토리를 base 폴더로 해서 vs-code 실행
code .

```
